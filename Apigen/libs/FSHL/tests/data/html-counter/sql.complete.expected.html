104: -- PERFORMANCE SCHEMA INSTALLATION
105: -- Note that this script is also reused by mysql_upgrade,
106: -- so we have to be very careful here to not destroy any
107: -- existing database named 'performance_schema' if it
108: -- can contain user data.
109: -- In case of downgrade, it's ok to drop unknown tables
110: -- from a future version, as long as they belong to the
111: -- performance schema engine.
112: --
113: 
114: set @have_old_pfs= (select count(*) from information_schema.schemata where schema_name='performance_schema');
115: 
116: SET @l1="SET @broken_tables = (select count(*) from information_schema.tables";
117: SET @l2=" where engine != \'PERFORMANCE_SCHEMA\' and table_schema=\'performance_schema\')";
